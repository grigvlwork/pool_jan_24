# Доступен файл для чтения: 17_(2).txt
lst = [int(x) for x in open('17_(2).txt']
c = 0
mx = 0
for i in range(len(lst) - 1):
  for j in range(i + 1, len(lst)):
    x = lst[i]
    y = lst[j]
    if ((x + y) % 18 == 0) ^ ((x * y) % 18 == 0):
      c += 1
      mx = max(mx, x + y)
print(c, mx)
